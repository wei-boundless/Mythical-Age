 971 |             const knownDist = toKnown.length();

 976 |               enemy.mesh.lookAt(knownPos);
 977 |             }
 978 |           }
 979 |         }

 981 |         
 982 |       case ENEMY_STATES.ATTACK:
 983 |         // 面向玩家
 984 |         enemy.mesh.lookAt(playerPos);

 986 |         if (enemy.attackCooldown <= 0) {

 996 |         if (distXZ > enemy.attackRange + 3) {
 997 |           enemy.state = ENEMY_STATES.CHASE;
 998 |           enemy.lastKnownPlayerPos = playerPos.clone();
 999 |           enemy.stateTimer = 0;
1000 |         }
1001 |         break;
1002 |         
1003 |       case ENEMY_STATES.LOST:
1004 |         // 最后位置巡逻
1005 |         if (canSeePlayer) {
1006 |           enemy.state = ENEMY_STATES.CHASE;
1007 |           enemy.lastKnownPlayerPos = playerPos.clone();
1008 |           enemy.stateTimer = 0;
1009 |         } else if (enemy.stateTimer > 4) {
1010 |           enemy.state = ENEMY_STATES.PATROL;

1012 |           enemy.stateTimer = 0;
1013 |         }
1014 |         break;
1015 |     }

1018 |     enemy.mesh.position.copy(enemyPos);

1020 |     // 碰撞检测
1021 |     const enemyBox = new THREE.Box3().setFromObject(enemy.mesh);
1022 |     for (const collider of allColliders) {

1039 |     enemy.mesh.position.y = 0;

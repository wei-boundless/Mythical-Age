 754 |   if (!playerState.alive || !controls.isLocked) return;

 764 |   
 765 |   const moveDir = new THREE.Vector3();
 766 |   if (keys['w']) moveDir.z -= 1;
 767 |   if (keys['s']) moveDir.z += 1;
 768 |   if (keys['a']) moveDir.x -= 1;
 769 |   if (keys['d']) moveDir.x += 1;
 770 |   
 771 |   if (moveDir.length() > 0) {
 772 |     moveDir.normalize();
 773 |     // 转换为世界方向
 774 |     const forward = new THREE.Vector3();
 775 |     camera.getWorldDirection(forward);
 776 |     forward.y = 0;
 777 |     forward.normalize();
 778 |     const right = new THREE.Vector3();
 779 |     right.crossVectors(forward, new THREE.Vector3(0, 1, 0)).normalize();
 780 |     
 781 |     const worldMove = new THREE.Vector3();
 782 |     worldMove.addScaledVector(forward, -moveDir.z);
 783 |     worldMove.addScaledVector(right, moveDir.x);
 784 |     worldMove.normalize();
 785 |     
 786 |     playerState.velocity.x = worldMove.x * speed;
 787 |     playerState.velocity.z = worldMove.z * speed;
 788 |   } else {
 789 |     // 摩擦力
 790 |     playerState.velocity.x *= 0.85;
 791 |     playerState.velocity.z *= 0.85;
 792 |   }
 793 |   
 794 |   // 跳跃
 795 |   if (keys[' '] && playerState.onGround) {
 796 |     playerState.velocity.y = JUMP_FORCE;
 797 |     playerState.onGround = false;
 798 |   }
 799 |   
 800 |   // 碰撞检测与位置更新
 801 |   const pos = camera.position.clone();
 802 |   
 803 |   // 水平移动（碰撞）
 804 |   const horizontalMove = new THREE.Vector3(
 805 |     playerState.velocity.x * dt,
 806 |     0,
 807 |     playerState.velocity.z * dt
 808 |   );
 809 |   
 810 |   const newPos = pos.clone().add(horizontalMove);
 811 |   const playerBox = new THREE.Box3(
 812 |     new THREE.Vector3(newPos.x - PLAYER_RADIUS, pos.y, newPos.z - PLAYER_RADIUS),
 813 |     new THREE.Vector3(newPos.x + PLAYER_RADIUS, pos.y + PLAYER_HEIGHT, newPos.z + PLAYER_RADIUS)
 814 |   );

 818 |   let collisionZ = false;
 819 |   
 820 |   for (const collider of allColliders) {

 822 |       // 轴向分离

 875 | // ===== 敌人 AI =====
 876 | function updateEnemyAI(deltaTime) {
 877 |   if (!playerState.alive) return;
 878 |   
 879 |   const playerPos = new THREE.Vector3();
 880 |   camera.getWorldPosition(playerPos);
 881 |   const playerDir = new THREE.Vector3();
 882 |   camera.getWorldDirection(playerDir);
 883 |   
 884 |   for (const enemy of enemies) {
 885 |     const dt = Math.min(deltaTime, 0.1);
 886 |     const enemyPos = enemy.mesh.position.clone();
 887 |     const distToPlayer = enemyPos.distanceTo(playerPos);
 888 |     
 889 |     // 更新攻击冷却
 890 |     enemy.attackCooldown = Math.max(0, enemy.attackCooldown - dt);
 891 |     enemy.stateTimer += dt;
 892 |     
 893 |     // 检测玩家
 894 |     const toPlayer = playerPos.clone().sub(enemyPos);
 895 |     toPlayer.y = 0;
 896 |     const distXZ = toPlayer.length();
 897 |     
 898 |     let canSeePlayer = false;
 899 |     if (distXZ < enemy.detectionRange) {
 900 |       // 视线检测
 901 |       const lookDir = new THREE.Vector3(0, 0, -1);
 902 |       lookDir.applyQuaternion(enemy.mesh.quaternion);
 903 |       lookDir.y = 0;
 904 |       lookDir.normalize();
 905 |       
 906 |       const toPlayerDir = toPlayer.clone().normalize();
 907 |       const dot = lookDir.dot(toPlayerDir);
 908 |       
 909 |       if (dot > Math.cos(enemy.fov / 2) || distXZ < 3) {
 910 |         // 射线检测有无遮挡
 911 |         const rayOrigin = enemyPos.clone();
 912 |         rayOrigin.y += 1.2;
 913 |         const rayDir = playerPos.clone().sub(rayOrigin).normalize();
 914 |         const raycaster = new THREE.Raycaster(rayOrigin, rayDir, 0, distToPlayer + 1);
 915 |         const obstaclesForRay = [];
 916 |         mapGroup.children.forEach(c => {
 917 |           if (c !== floor && c !== ceiling && !c.userData.enemyId) {
 918 |             obstaclesForRay.push(c);
 919 |           }
 920 |         });
 921 |         const intersects = raycaster.intersectObjects(obstaclesForRay);
 922 |         canSeePlayer = intersects.length === 0 || intersects[0].distance > distToPlayer - 0.5;
 923 |       }
 924 |     }
 925 |     
 926 |     // 状态机
 927 |     switch (enemy.state) {
 928 |       case ENEMY_STATES.PATROL:
 929 |         if (canSeePlayer) {
 930 |           enemy.state = ENEMY_STATES.CHASE;
 931 |           enemy.lastKnownPlayerPos = playerPos.clone();
 932 |           enemy.stateTimer = 0;
 933 |         } else {
 934 |           // 巡逻移动
 935 |           const toTarget = enemy.patrolTarget.clone().sub(enemyPos);
 936 |           const targetDist = toTarget.length();
 937 |           if (targetDist < 0.5) {
 938 |             enemy.patrolWaitTimer += dt;
 939 |             if (enemy.patrolWaitTimer > 2) {
 940 |               enemy.patrolTarget = getRandomPatrolPoint(enemyPos);
 941 |               enemy.patrolWaitTimer = 0;
 942 |             }
 943 |           } else {
 944 |             toTarget.normalize();
 945 |             const moveAmount = enemy.speed * 0.5 * dt;
 946 |             enemyPos.x += toTarget.x * moveAmount;
 947 |             enemyPos.z += toTarget.z * moveAmount;
 948 |             // 面向移动方向
 949 |             enemy.mesh.lookAt(enemyPos.x + toTarget.x, enemyPos.y, enemyPos.z + toTarget.z);
 950 |           }
 951 |         }
 952 |         break;
 953 |         
 954 |       case ENEMY_STATES.CHASE:
 955 |         if (!canSeePlayer && enemy.stateTimer > 3) {
 956 |           enemy.state = ENEMY_STATES.LOST;
 957 |           enemy.stateTimer = 0;
 958 |         } else if (canSeePlayer) {
 959 |           enemy.lastKnownPlayerPos = playerPos.clone();
 960 |           // 追击
 961 |           const chaseDir = toPlayer.clone().normalize();
 962 |           const moveAmount = enemy.speed * dt;
 963 |           if (distXZ > 1.8) {
 964 |             enemyPos.x += chaseDir.x * moveAmount;
 965 |             enemyPos.z += chaseDir.z * moveAmount;
 966 |           }
 967 |           enemy.mesh.lookAt(playerPos);
 968 |           
 969 |           // 进入攻击范围
 970 |           if (distXZ <= enemy.attackRange && enemy.attackCooldown <= 0) {
 971 |             enemy.state = ENEMY_STATES.ATTACK;
 972 |             enemy.stateTimer = 0;
 973 |           }
 974 |         } else {
 975 |           // 向最后已知位置移动
 976 |           const knownPos = enemy.lastKnownPlayerPos;
 977 |           if (knownPos) {
 978 |             const toKnown = knownPos.clone().sub(enemyPos);
 979 |             const knownDist = toKnown.length();
 980 |             if (knownDist > 0.5) {
 981 |               toKnown.normalize();
 982 |               enemyPos.x += toKnown.x * enemy.speed * dt;
 983 |               enemyPos.z += toKnown.z * enemy.speed * dt;
 984 |               enemy.mesh.lookAt(knownPos);
 985 |             }
 986 |           }
 987 |         }
 988 |         break;
 989 |         
 990 |       case ENEMY_STATES.ATTACK:
 991 |         // 面向玩家
 992 |         enemy.mesh.lookAt(playerPos);
 993 |         
 994 |         if (enemy.attackCooldown <= 0) {
 995 |           // 攻击
 996 |           damagePlayer(enemy.attackDamage);
 997 |           enemy.attackCooldown = 0.8 + Math.random() * 0.4;
 998 |           
 999 |           // 攻击特效
1000 |           createEnemyAttackEffect(enemyPos);
1001 |         }
1002 |         
1003 |         // 检查玩家是否跑远
1004 |         if (distXZ > enemy.attackRange + 3) {
1005 |           enemy.state = ENEMY_STATES.CHASE;
1006 |           enemy.lastKnownPlayerPos = playerPos.clone();
1007 |           enemy.stateTimer = 0;
1008 |         }
1009 |         break;
1010 |         
1011 |       case ENEMY_STATES.LOST:
1012 |         // 最后位置巡逻
1013 |         if (canSeePlayer) {
1014 |           enemy.state = ENEMY_STATES.CHASE;
1015 |           enemy.lastKnownPlayerPos = playerPos.clone();
1016 |           enemy.stateTimer = 0;
1017 |         } else if (enemy.stateTimer > 4) {
1018 |           enemy.state = ENEMY_STATES.PATROL;
1019 |           enemy.patrolTarget = getRandomPatrolPoint(enemyPos);
1020 |           enemy.stateTimer = 0;
1021 |         }
1022 |         break;
1023 |     }
1024 |     
1025 |     // 应用位置并碰撞
1026 |     enemy.mesh.position.copy(enemyPos);
1027 |     
1028 |     // 碰撞检测
1029 |     const enemyBox = new THREE.Box3().setFromObject(enemy.mesh);
1030 |     for (const collider of allColliders) {
1031 |       if (enemyBox.intersectsBox(collider)) {
1032 |         // 简单推开
1033 |         const center = enemyPos.clone();
1034 |         const colliderCenter = new THREE.Vector3();
1035 |         collider.getCenter(colliderCenter);
1036 |         const pushDir = center.sub(colliderCenter);
1037 |         pushDir.y = 0;
1038 |         if (pushDir.length() < 0.01) pushDir.set(Math.random() - 0.5, 0, Math.random() - 0.5);
1039 |         pushDir.normalize();
1040 |         enemy.mesh.position.addScaledVector(pushDir, 0.1);
1041 |       }
1042 |     }
1043 |     
1044 |     // 边界限制
1045 |     enemy.mesh.position.x = Math.max(-28, Math.min(28, enemy.mesh.position.x));
1046 |     enemy.mesh.position.z = Math.max(-28, Math.min(28, enemy.mesh.position.z));
1047 |     enemy.mesh.position.y = 0;
1048 |   }
1049 | }

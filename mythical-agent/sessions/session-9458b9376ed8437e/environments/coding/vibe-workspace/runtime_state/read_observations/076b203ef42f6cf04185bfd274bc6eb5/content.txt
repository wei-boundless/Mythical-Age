574 |   if (playerState.fireCooldown > 0) playerState.fireCooldown -= deltaTime;
575 |   if (playerState.reloading) {
576 |     playerState.reloadTimer -= deltaTime;
577 |     if (playerState.reloadTimer <= 0) finishReload();
578 |   }
579 | 
580 |   // 玩家移动
581 |   const moveDir = new THREE.Vector3();
582 |   const forward = new THREE.Vector3();
583 |   camera.getWorldDirection(forward);
584 |   forward.y = 0;
585 |   forward.normalize();
586 |   const right = new THREE.Vector3();
587 |   right.crossVectors(forward, new THREE.Vector3(0, 1, 0)).normalize();
588 | 
589 |   if (keys['KeyW']) moveDir.add(forward);
590 |   if (keys['KeyS']) moveDir.sub(forward);
591 |   if (keys['KeyA']) moveDir.sub(right);
592 |   if (keys['KeyD']) moveDir.add(right);
593 | 
594 |   if (moveDir.length() > 0) {
595 |     moveDir.normalize();
596 |     const newX = camera.position.x + moveDir.x * PLAYER_SPEED * deltaTime;
597 |     const newZ = camera.position.z + moveDir.z * PLAYER_SPEED * deltaTime;
598 | 
599 |     const collision = checkObstacleCollision(newX, newZ, PLAYER_RADIUS);
600 |     if (collision.hit) {
601 |       camera.position.x += moveDir.x * PLAYER_SPEED * deltaTime + collision.pushX;
602 |       camera.position.z += moveDir.z * PLAYER_SPEED * deltaTime + collision.pushZ;
603 |     } else {
604 |       camera.position.x = newX;
605 |       camera.position.z = newZ;
606 |     }
607 |   }
608 | 
609 |   // 重力与跳跃
610 |   if (!playerState.onGround) {
611 |     playerState.velocity.y -= GRAVITY * deltaTime;
612 |   }
613 |   camera.position.y += playerState.velocity.y * deltaTime;
614 |   if (camera.position.y <= PLAYER_HEIGHT) {
615 |     camera.position.y = PLAYER_HEIGHT;
616 |     playerState.velocity.y = 0;
617 |     playerState.onGround = true;

621 |     playerState.velocity.y = JUMP_VELOCITY;
622 |     playerState.onGround = false;
623 |   }
624 | 
625 |   // 限制玩家在地图范围内
626 |   const MAP_BOUND = 48;
627 |   camera.position.x = Math.max(-MAP_BOUND, Math.min(MAP_BOUND, camera.position.x));
628 |   camera.position.z = Math.max(-MAP_BOUND, Math.min(MAP_BOUND, camera.position.z));
629 | 
630 |   // 更新敌人
631 |   for (let i = enemies.length - 1; i >= 0; i--) {
632 |     const enemy = enemies[i];

634 |       camera.position.x - enemy.position.x,
635 |       0,
636 |       camera.position.z - enemy.position.z
637 |     );
638 |     const dist = toPlayer.length();
639 | 
640 |     if (dist > 0.8) {
641 |       toPlayer.normalize();
642 |       const moveX = toPlayer.x * enemy.userData.speed * deltaTime;
643 |       const moveZ = toPlayer.z * enemy.userData.speed * deltaTime;
644 |       if (!checkEnemyObstacleCollision(enemy, moveX, moveZ)) {
645 |         enemy.position.x += moveX;
646 |         enemy.position.z += moveZ;
647 |       }
648 |       enemy.lookAt(camera.position.x, enemy.position.y, camera.position.z);
649 |     }
650 | 
651 |     // 攻击玩家
652 |     if (dist < 1.5) {
653 |       enemy.userData.attackCooldown -= deltaTime;
654 |       if (enemy.userData.attackCooldown <= 0) {
655 |         damagePlayer(enemy.userData.damage);
656 |         enemy.userData.attackCooldown = enemy.userData.attackRate;
657 |       }
658 |     }
659 |   }

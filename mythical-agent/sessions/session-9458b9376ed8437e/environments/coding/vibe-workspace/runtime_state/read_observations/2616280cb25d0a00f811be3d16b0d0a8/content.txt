 481 |     while (hitObj && !hitObj.userData.enemyId) {
 482 |       hitObj = hitObj.parent;
 483 |     }
 484 |     if (hitObj && hitObj.userData.enemyId) {
 485 |       const enemyId = hitObj.userData.enemyId;
 486 |       const enemy = enemies.find(e => e.id === enemyId);
 487 |       if (enemy) {
 488 |         // 判断是否爆头
 489 |         const isHeadshot = intersects[0].object.userData.isHead === true;
 490 |         const damage = isHeadshot ? weapon.damage * weapon.headMultiplier : weapon.damage;
 491 |         enemy.health -= damage;
 492 |         enemy.lastHitTime = performance.now() / 1000;
 493 |         
 494 |         // 击中特效
 495 |         createHitEffect(intersects[0].point, isHeadshot);
 496 |         
 497 |         if (enemy.health <= 0) {
 498 |           killEnemy(enemy, isHeadshot);
 499 |         }
 500 |       }
 501 |     }
 502 |   }
 503 |   
 504 |   // 枪口闪光
 505 |   createMuzzleFlash();
 506 |   
 507 |   return true;
 508 | }

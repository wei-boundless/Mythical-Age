310 | // ==================== 碰撞检测 ====================
311 | function checkObstacleCollision(x, z, radius) {
312 |   for (const obs of obstacles) {
313 |     const dx = x - obs.position.x;
314 |     const dz = z - obs.position.z;
315 |     const dist = Math.sqrt(dx * dx + dz * dz);
316 |     if (dist < obs.userData.radius + radius) {
317 |       return { hit: true, pushX: dx / dist * (obs.userData.radius + radius - dist), pushZ: dz / dist * (obs.userData.radius + radius - dist) };
318 |     }
319 |   }
320 |   return { hit: false };
321 | }
322 | 
323 | function checkEnemyObstacleCollision(enemy, dx, dz) {
324 |   const newX = enemy.position.x + dx;
325 |   const newZ = enemy.position.z + dz;
326 |   for (const obs of obstacles) {
327 |     const ox = newX - obs.position.x;
328 |     const oz = newZ - obs.position.z;
329 |     if (Math.sqrt(ox * ox + oz * oz) < obs.userData.radius + 0.5) return true;
330 |   }
331 |   // 检查与其他敌人的碰撞
332 |   for (const other of enemies) {
333 |     if (other === enemy) continue;
334 |     const ex = newX - other.position.x;
335 |     const ez = newZ - other.position.z;
336 |     if (Math.sqrt(ex * ex + ez * ez) < 0.8) return true;
337 |   }
338 |   return false;
339 | }

396 | // ==================== 方案B：更新准心与镜头 ====================
397 | function updateCrosshairAndCamera(deltaTime) {
398 |   // 准心自由移动 —— 使用鼠标累积偏移
399 |   // mouseDX / mouseDY 在 mousemove 事件中累积
400 | 
401 |   // 读取并清零累积偏移
402 |   const dx = window._mouseDX || 0;
403 |   const dy = window._mouseDY || 0;
404 |   window._mouseDX = 0;
405 |   window._mouseDY = 0;
406 | 
407 |   // 更新准心屏幕位置
408 |   crosshairState.screenX += dx;
409 |   crosshairState.screenY += dy;
410 | 
411 |   // 钳制到屏幕范围内
412 |   crosshairState.screenX = Math.max(0, Math.min(window.innerWidth, crosshairState.screenX));
413 |   crosshairState.screenY = Math.max(0, Math.min(window.innerHeight, crosshairState.screenY));
414 | 
415 |   // 更新准心 DOM 元素位置
416 |   crosshairEl.style.left = crosshairState.screenX + 'px';
417 |   crosshairEl.style.top = crosshairState.screenY + 'px';
418 |   crosshairEl.style.transform = 'translate(-50%, -50%)';
419 | 
420 |   // 计算准心偏离屏幕中心的比例
421 |   const halfW = window.innerWidth / 2;
422 |   const halfH = window.innerHeight / 2;
423 |   crosshairState.offsetX = (crosshairState.screenX - halfW) / halfW;
424 |   crosshairState.offsetY = (crosshairState.screenY - halfH) / halfH;
425 | 
426 |   // 镜头边缘转动：准心靠近边缘时转动相机
427 |   const edgeX = crosshairState.screenX;
428 |   const edgeY = crosshairState.screenY;
429 | 
430 |   let rotateH = 0;
431 |   let rotateV = 0;
432 | 
433 |   if (edgeX < EDGE_MARGIN) {
434 |     rotateH = -EDGE_ROTATE_SPEED * deltaTime * (1 - edgeX / EDGE_MARGIN);
435 |   } else if (edgeX > window.innerWidth - EDGE_MARGIN) {
436 |     rotateH = EDGE_ROTATE_SPEED * deltaTime * (1 - (window.innerWidth - edgeX) / EDGE_MARGIN);
437 |   }
438 | 
439 |   if (edgeY < EDGE_MARGIN) {
440 |     rotateV = EDGE_ROTATE_SPEED * deltaTime * (1 - edgeY / EDGE_MARGIN);
441 |   } else if (edgeY > window.innerHeight - EDGE_MARGIN) {
442 |     rotateV = -EDGE_ROTATE_SPEED * deltaTime * (1 - (window.innerHeight - edgeY) / EDGE_MARGIN);
443 |   }
444 | 
445 |   // 应用镜头旋转
446 |   if (rotateH !== 0) {
447 |     camera.rotateY(-rotateH);
448 |   }
449 |   if (rotateV !== 0) {
450 |     // 限制垂直旋转角度
451 |     const currentRotX = camera.rotation.x;
452 |     const newRotX = currentRotX + rotateV;
453 |     if (newRotX > -Math.PI / 2.2 && newRotX < Math.PI / 2.2) {
454 |       camera.rotateX(rotateV);
455 |     }
456 |   }
457 | }

 308 | // 每帧在 animate 中消费鼠标 delta，更新准心位置并检测边缘旋转
 309 | function updateCrosshairAndCamera(deltaTime) {
 310 |   // 1. 用鼠标 delta 移动准心
 311 |   crosshairX += mouseDX;
 312 |   crosshairY += mouseDY;
 313 |   mouseDX = 0;
 314 |   mouseDY = 0;
 315 | 
 316 |   // 2. 钳制准心在屏幕内
 317 |   crosshairX = Math.max(0, Math.min(window.innerWidth, crosshairX));
 318 |   crosshairY = Math.max(0, Math.min(window.innerHeight, crosshairY));
 319 | 
 320 |   // 3. 更新准心 DOM 位置
 321 |   const ch = document.getElementById('crosshair');
 322 |   ch.style.left = (crosshairX - 12) + 'px';
 323 |   ch.style.top = (crosshairY - 12) + 'px';
 324 | 
 325 |   // 4. 边缘检测 - 准心靠近屏幕边缘时转动镜头
 326 |   const dt = Math.min(deltaTime, 0.1); // 防止大帧间隔
 327 |   let rotateX = 0, rotateY = 0;
 328 | 
 329 |   if (crosshairX < EDGE_THRESHOLD) {
 330 |     rotateY = -EDGE_ROTATE_SPEED * dt * (1 - crosshairX / EDGE_THRESHOLD);
 331 |   } else if (crosshairX > window.innerWidth - EDGE_THRESHOLD) {
 332 |     rotateY = EDGE_ROTATE_SPEED * dt * (1 - (window.innerWidth - crosshairX) / EDGE_THRESHOLD);
 333 |   }
 334 | 
 335 |   if (crosshairY < EDGE_THRESHOLD) {
 336 |     rotateX = -EDGE_ROTATE_SPEED * dt * (1 - crosshairY / EDGE_THRESHOLD);
 337 |   } else if (crosshairY > window.innerHeight - EDGE_THRESHOLD) {
 338 |     rotateX = EDGE_ROTATE_SPEED * dt * (1 - (window.innerHeight - crosshairY) / EDGE_THRESHOLD);
 339 |   }
 340 | 
 341 |   // 5. 应用镜头旋转
 342 |   if (rotateX !== 0 || rotateY !== 0) {
 343 |     viewEuler.setFromQuaternion(camera.quaternion);
 344 |     viewEuler.y -= rotateY * MOUSE_SENSITIVITY * 60; // 归一化到约 60fps 手感
 345 |     viewEuler.x -= rotateX * MOUSE_SENSITIVITY * 60;
 346 |     viewEuler.x = Math.max(MIN_PITCH, Math.min(MAX_PITCH, viewEuler.x));
 347 |     camera.quaternion.setFromEuler(viewEuler);
 348 |   }
